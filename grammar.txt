START
	| STMTS

STMTS
	| STMTS STMT
	| STMT

STMT
	| { STMTS }
	| IF ( EXPR ) STMT
	| IF ( EXPR ) STMT ELSE STMT
	| WHILE ( EXPR ) STMT
	| VAR ID TYPE ;
	| ID = EXPR ;
	| PRINT EXPR ;

EXPR
	| ( EXPR ) 
	| EXPR && EXPR
	| EXPR || EXPR
	| EXPR == EXPR
	| EXPR != EXPR
	| EXPR >= EXPR 
	| EXPR >  EXPR
	| EXPR <= EXPR
	| EXPR <  EXPR
	| EXPR *  EXPR
	| EXPR /  EXPR
	| EXPR +  EXPR
	| EXPR -  EXPR
	| ID
	| NUM

TYPE
	| INT
	| FLOAT
